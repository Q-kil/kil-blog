function setLaunch() {
  var e = document.getElementById("launch");
  if (e) {
    var n = document.documentElement
      , o = e.clientWidth
      , r = e.clientHeight
      , t = n.getBoundingClientRect()
      , i = t.width
      , a = t.height;
    o > i && (e.style.width = i + "px"),
    r != a && (e.style.height = a + "px")
  }
}
